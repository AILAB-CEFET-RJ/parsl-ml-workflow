def create_baseline_model():
    model = Sequential()
    model.add(Dense(250, input_dim=10, activation='relu'))
    model.add(Dense(125, activation='relu'))
    model.add(Dense(1, activation='linear'))
    model.compile(loss='mse', optimizer='adam', metrics=['mse'])
    
    return model
